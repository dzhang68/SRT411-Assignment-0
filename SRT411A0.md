``` r
fun11.3 = function(arg1)
{
  s = c()
  for(i in (1:length(arg1)))
  {
    if(arg1[i] < 5 | arg1[i] > 90) 
    {
      s[i] = arg1[i]  * 10
    }
    else
    {
      s[i] = arg1[i] * 0.1
```
